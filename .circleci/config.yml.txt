version: 2.1
jobs:
  test:
    docker:
      - image: cimg/python:3.8
    steps:
      - checkout
      - run:
          name: Run Python script test
          command: python3 hello.py

  deploy:
    docker:
      - image: cimg/python:3.8
    steps:
      - checkout
      - run:
          name: Install Ansible
          command: |
            python3 -m pip install --upgrade pip
            pip install ansible
      - run:
          name: Run Ansible playbook to deploy
          command: |
            ansible-playbook -i hosts.ini deploy.yml --private-key ~/.ssh/id_rsa

workflows:
  version: 2
  test_and_deploy:
    jobs:
      - test
      - deploy:
          requires:
            - test
